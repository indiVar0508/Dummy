def greet(name):
    """The method greets you with your name with random probability

    Examples:
        >>> greet("Indi")
        Hi, Indi

    Args:
        name: Your name

    Returns:
        Greeting message
    """
    chance = random.random()
    print(f"Hi, {name}" if chance < 0.2 else f"Hola, {name}" if (chance > 0.5) and (chance < 0.8) else f"Bojour, {name}")
